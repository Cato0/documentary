class Point {
    
    constructor (private _x?: number, private _y?: number) {        # Add the ? if this input is optional
        this.x = x;
        this.y = y;
    }
    
    get x() {
        return this._x;
    }
    
    set x(value) {
        this._x = value
    }
    
    draw () {
        console.log(this.x);
    }
}

let p1 = new Point();
